🔗 Google Sheets 연동 설정
아래 안내에 따라 Google Sheets에 Apps Script를 연결하면, 평가 데이터가 자동으로 스프레드시트에 저장됩니다.
⚠️ 아직 연동 전입니다. 현재는 브라우저 로컬 저장소만 사용 중입니다.
📋 Google Sheets 연동 방법 (단계별 안내)
① Google Sheets 생성
drive.google.com → 새 스프레드시트 생성 → 이름: "계약직 면접평가 결과"

② Apps Script 열기
확장 프로그램 → Apps Script 클릭

③ 아래 코드 전체 붙여넣기 (기존 코드 삭제 후)
var SHEET_ID = '18dxEd1Ie8LAEVSAMfgCM0Kks-TqJnq6nbTz0KW5Q_-0'; function doPost(e) { var ss = SpreadsheetApp.openById(SHEET_ID); var sheet = ss.getSheets()[0]; var data = {}; try { if (e.parameter && e.parameter.data) { data = JSON.parse(e.parameter.data); } else if (e.postData) { data = JSON.parse(e.postData.contents); } } catch(err) {} sheet.appendRow([ new Date(), data.projectTitle || '', data.applicantName || '', data.jobType || '', data.department || '', data.evaluatorName || '', data.evalDate || '', data.score1 || 0, data.score2 || 0, data.score3 || 0, data.score4 || 0, data.score5 || 0, data.totalScore || 0, data.bohunPoint || 0, data.finalScore || 0, data.grade || '', data.comment || '' ]); return ContentService.createTextOutput(JSON.stringify({result:'ok'})).setMimeType(ContentService.MimeType.JSON); } function doGet(e) { return ContentService.createTextOutput(JSON.stringify({status:'ok'})).setMimeType(ContentService.MimeType.JSON); }
④ 배포
배포 → 새 배포 → 유형: 웹 앱 → 액세스: 모든 사용자 → 배포

⑤ 웹앱 URL 복사 후 위 입력란에 붙여넣기
